# Data Model

## Result (Deliverable)

```typescript
interface Result {
  id: string;                    // UUID
  title: string;                 // Short action-led title
  content: string;               // Full markdown/text body
  format: "markdown" | "text" | "code" | "json" | "table";
  
  trust_level: "authoritative" | "conclusion" | "exploratory";
  confidence: number;            // 0-1
  source_trust: "agent_authoritative" | "agent_conclusion" | 
                "agent_exploratory" | "operator_authoritative";
  
  agent_id: string;              // Who created it
  project_id: string;
  
  source_urls: string[];         // Where facts come from
  reasoning: string;             // Why this is true
  file_ids: string[];            // Attached artifacts
  
  created_at: ISO8601;
  updated_at: ISO8601;
  
  supersedes: string;            // Prior result id (if update)
  relation_type: "updates" | "extends" | "derives" | "contradicts";
  
  pinned: boolean;               // Operator pinned this
}
```

## Decision (Directive/Rule)

```typescript
interface Decision {
  id: string;
  
  statement: string;             // The decision
  kind: "directive" | "rule" | "conclusion";
  rationale: string;             // Why
  
  status: "open" | "done" | "superseded";
  
  agent_id: string;              // Who filed it
  project_id: string;
  
  targets: DecisionScope;        // Who it applies to
  effective_from: ISO8601;
  expires_at?: ISO8601;
  
  created_at: ISO8601;
  updated_at: ISO8601;
  
  refs: string[];                // For conclusions: what it closes
}
```

## DecisionScope

```typescript
interface DecisionScope {
  kind: "agent" | "project" | "surface" | "all";
  agent_ids?: string[];          // For agent scope
  project?: string;              // For project scope
  surfaces?: string[];           // For surface scope (claude-code, etc)
}
```

## File (Artifact)

```typescript
interface File {
  file_id: string;               // UUID
  
  name: string;                  // Filename with extension
  mime: string;                  // Content type
  size: number;                  // Bytes
  
  ref: string;                   // Stable slug for versioning
  version: number;               // v1, v2, v3...
  
  upload_url: string;            // For signed uploads (init response only)
  
  agent_id: string;
  project_id: string;
  
  created_at: ISO8601;
  updated_at: ISO8601;
}
```

## Table (Persistent Data)

```typescript
interface Table {
  table_ref: string;             // Stable slug
  version: number;               // v1, v2, v3...
  
  title: string;
  columns: string[];
  rows: Record<string, any>[];
  
  trust_level: "authoritative" | "conclusion";
  
  agent_id: string;
  project_id: string;
  
  note_above: string;            // Markdown
  note_below: string;
  
  created_at: ISO8601;
  updated_at: ISO8601;
  row_count: number;
}
```

## Event (Activity Log)

```typescript
interface Event {
  id: string;
  
  message: string;               // One-line activity summary
  kind: "log" | "milestone" | "system";
  
  agent_id: string;
  project_id: string;
  
  timestamp: ISO8601;
}
```

## Agent (Runtime)

```typescript
interface Agent {
  id: string;                    // Stable slug
  name: string;                  // Display name
  type: "code" | "cowork" | "scheduled" | "custom";
  project: string;               // Project slug
  
  provider?: string;             // anthropic, openai, etc
  surface?: string;              // claude-code, cowork, api, custom
  session_id?: string;           // External session tracker
  model_family?: string;         // claude-haiku-4-5, gpt-4, etc
  
  last_run: ISO8601;             // Latest heartbeat
  status: "running" | "idle" | "error";
  
  events: Event[];               // Activity timeline
  results: Result[];             // Deliverables
  decisions: Decision[];         // Filed decisions
  files: File[];                 // Uploaded artifacts
  tables: Table[];               // Tables maintained
}
```

## Heartbeat Request

```typescript
interface HeartbeatRequest {
  agent: {
    id: string;
    project: string;
    type?: "code" | "cowork" | "scheduled" | "custom";
    name?: string;
    provider?: string;
    surface?: string;
    session_id?: string;
    model_family?: string;
  };
  status: "running" | "idle" | "error";
  context?: string;              // What you're doing (for running)
  summary?: string;              // Run summary (for idle)
}
```

## Timestamps

All timestamps are ISO 8601 format:
```
2026-06-28T14:30:45Z
```

## Field Validation Rules

| Field | Rules |
|-------|-------|
| `agent.id` | Kebab-case, 1-120 chars, stable |
| `agent.project` | Kebab-case, 1-160 chars, tool-neutral (no provider names) |
| `title` | 1-80 chars, start with past-tense verb |
| `content` | Non-empty, max ~100KB |
| `trust_level` | One of enum values |
| `confidence` | 0.0-1.0 (capped by trust tier) |
| `statement` | Imperative, 1-2000 chars |
| `mime` | Valid MIME type |
| `file_size` | 1-50MB |
| `ref` | Lowercase slug, no spaces, reusable |
| `table_ref` | Lowercase slug, allows `/` for namespace |
| `columns` | 1-15 columns, unique names |
| `rows` | Max 500 per write, objects with string keys |
